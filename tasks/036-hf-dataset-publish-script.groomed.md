## Scope

Upload `runs/main/results.parquet` and `runs/niah/main/results.csv` to a HuggingFace dataset repo (`Helmo21/asia-fertility`) so downstream consumers can `load_dataset(...)` without cloning the GitHub repo. Includes a one-time `huggingface-cli login` for first release; subsequent releases automate via `HF_TOKEN`.

### Interface and contract

`scripts/publish_hf_dataset.py`:

```python
"""Publish a fertiscope run's artifacts to the HuggingFace dataset hub.

Usage:
    python scripts/publish_hf_dataset.py \
      --run runs/main \
      --niah-run runs/niah/main \
      --repo Helmo21/asia-fertility \
      --version v0.3.0
"""
from __future__ import annotations
import argparse
import json
import os
import sys
from pathlib import Path
from typing import Optional


def publish(
    run_dir: Path,
    niah_run_dir: Optional[Path],
    repo_id: str,
    version: str,
    *,
    dry_run: bool = False,
) -> None:
    try:
        from huggingface_hub import HfApi, login   # type: ignore
    except ImportError:
        sys.exit("Missing huggingface_hub. Install with: pip install fertiscope[hf]")

    if not os.environ.get("HF_TOKEN"):
        sys.exit("HF_TOKEN env var required. Get a token from https://huggingface.co/settings/tokens (write scope).")

    api = HfApi(token=os.environ["HF_TOKEN"])

    # Verify the repo exists
    try:
        api.repo_info(repo_id, repo_type="dataset")
    except Exception:
        sys.exit(f"Dataset repo '{repo_id}' not found. Create it manually first at https://huggingface.co/new-dataset.")

    # Load manifest for commit message provenance
    manifest = json.loads((run_dir / "manifest.json").read_text("utf-8"))
    config_sha = manifest.get("config_sha256", "unknown")[:12]
    commit_msg = f"{version} â€” manifest {config_sha}"
    if dry_run:
        print(f"[DRY] Would commit: {commit_msg}")

    # Files to upload, with paths inside the repo
    uploads: list[tuple[Path, str]] = [
        (run_dir / "results.parquet", f"{version}/results.parquet"),
        (run_dir / "results.csv",     f"{version}/results.csv"),
        (run_dir / "manifest.json",   f"{version}/manifest.json"),
        (run_dir / "leaderboard.json",f"{version}/leaderboard.json"),
    ]
    if niah_run_dir is not None:
        if (niah_run_dir / "results.csv").exists():
            uploads.append((niah_run_dir / "results.csv", f"{version}/niah_results.csv"))

    # Also upload as "latest" so consumers can pin or follow
    uploads += [(src, dst.replace(version + "/", "latest/")) for src, dst in uploads]

    for src, dst in uploads:
        if not src.exists():
            print(f"  âš  skipping (missing): {src}")
            continue
        if dry_run:
            print(f"  [DRY] {src} â†’ {dst}")
            continue
        print(f"  â†‘ {src.name} â†’ {dst}")
        api.upload_file(
            path_or_fileobj=str(src),
            path_in_repo=dst,
            repo_id=repo_id,
            repo_type="dataset",
            commit_message=commit_msg,
        )

    print(f"âś“ Published to https://huggingface.co/datasets/{repo_id} (version: {version})")


def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--run", required=True, type=Path)
    p.add_argument("--niah-run", type=Path, default=None)
    p.add_argument("--repo", required=True)
    p.add_argument("--version", required=True, help="Tag-like, e.g. v0.3.0")
    p.add_argument("--dry-run", action="store_true")
    args = p.parse_args()
    publish(args.run, args.niah_run, args.repo, args.version, dry_run=args.dry_run)


if __name__ == "__main__":
    main()
```

`.github/workflows/publish_hf.yml` â€” manually triggered:

```yaml
name: Publish HF dataset

on:
  workflow_dispatch:
    inputs:
      version:
        description: "Version tag (e.g. v0.3.0)"
        required: true
      run_dir:
        description: "Main run directory inside the runner workspace"
        default: "runs/main"
      niah_run_dir:
        description: "NIAH run directory (or empty)"
        default: "runs/niah/main"

jobs:
  publish:
    runs-on: ubuntu-latest
    environment: hf-dataset
    steps:
      - uses: actions/checkout@v4
      - uses: astral-sh/setup-uv@v3
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - run: uv sync --extra hf
      - name: Download artifacts (if any)
        run: |
          # In practice, the maintainer attaches release artifacts here.
          # For now: read locally-checked-in artifacts.
          ls ${{ inputs.run_dir }} || true
      - name: Publish
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: |
          uv run python scripts/publish_hf_dataset.py \
            --run ${{ inputs.run_dir }} \
            --niah-run ${{ inputs.niah_run_dir }} \
            --repo Helmo21/asia-fertility \
            --version ${{ inputs.version }}
```

### Notes

- HF dataset repo must be **created manually** the first time (one-click UI at `huggingface.co/new-dataset`).
- The script is idempotent: re-uploading the same version overwrites files in the `{version}/` subdir + the `latest/` mirror.
- `HF_TOKEN` is the ONE secret stored as a GitHub Actions secret (Settings â†’ Secrets and variables â†’ Actions). Scope: write to `Helmo21/asia-fertility`.
- The `latest/` mirror lets downstream `load_dataset("Helmo21/asia-fertility", "leaderboard")` always get the most recent without specifying a version.
- The HF dataset README has YAML frontmatter that controls HF's dataset viewer.
